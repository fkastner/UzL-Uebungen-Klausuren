\begin{aufgabe}[8 Punkte]
    Kreuzen Sie bitte an, ob die jeweilige Aussage wahr (W) oder falsch (F) ist. F\"ur jedes
    korrekt gesetzte Kreuz gibt es 0.5 Punkte, f\"ur jedes nicht korrekt gesetzte Kreuz werden 0.5 Punkte
    abgezogen. Ist die Summe der erzielten Punkte negativ, so wird die gesamte Aufgabe mit
    0 Punkten bewertet.
    \begin{enumerate}[a)]
        \item Frage

        \begin{ankreuzblock}
            \antwort{W}{Antwort}
            \antwort{F}{Antwort}
            \antwort{F}{Antwort}
            \antwort{W}{Antwort}
        \end{ankreuzblock}
        \item Frage

        \begin{ankreuzblock}
            \antwort{W}{Antwort}
            \antwort{F}{Antwort}
            \antwort{F}{Antwort}
            \antwort{W}{Antwort}
        \end{ankreuzblock}
        \item Frage

        \begin{ankreuzblock}
            \antwort{W}{Antwort}
            \antwort{F}{Antwort}
            \antwort{F}{Antwort}
            \antwort{W}{Antwort}
        \end{ankreuzblock}
        \item Frage

        \begin{ankreuzblock}
            \antwort{W}{Antwort}
            \antwort{F}{Antwort}
            \antwort{F}{Antwort}
            \antwort{W}{Antwort}
        \end{ankreuzblock}
    \end{enumerate}
\end{aufgabe}
